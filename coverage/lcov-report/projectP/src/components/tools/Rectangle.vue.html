<template>
  <li id="tool-square" @click="Rectangle()">
    <i class="far fa-square tool square-icon" id="square"></i>
  </li>
</template>
 
<script>
/* eslint-disable */
export default {
  name: 'Rectangle',
  props: {
    isActive: {
      type: Boolean,
      default: true,
    },
  },
  data() {
    return {};
  },
  methods: {
    Rectangle() {
      const canvas = document.getElementById('canvas');
      const ctx = canvas.getContext('2d');
      // Variables
      const canvasx = $(canvas).offset().left;
      const canvasy = $(canvas).offset().top;
      let last_mousex = 0;
      let last_mousey = 0;
      let mousex = 0;
      let mousey = 0;
      let mousedown = false;
 
      $(canvas).on('mousedown', e => {
        last_mousex = e.clientX - canvasx;
        last_mousey = e.clientY - canvasy;
        mousedown = true;
      });
 
      $(canvas).on('mouseup', e => {
        mousedown = false;
      });
 
      $(canvas).on('mousemove', e => {
        mousex = e.clientX - canvasx;
        mousey = e.clientY - canvasy;
        if (mousedown) {
          ctx.clearRect(0, 0, canvas.width, canvas.height); // clear canvas
          ctx.beginPath();
          const width = mousex - last_mousex;
          const height = mousey - last_mousey;
          ctx.rect(last_mousex, last_mousey, width, height);
          ctx.strokeStyle = 'black';
          ctx.lineWidth = 10;
          ctx.stroke();
        }
      });
    },
  },
};
</script>
